MrSheila!


Make sure to add an App.config file with an oauth key like so:

```
<?xml version="1.0" encoding="utf-8" ?>
<configuration>
    <startup> 
        <supportedRuntime version="v4.0" sku=".NETFramework,Version=v4.5.2" />
    </startup>

  <appSettings>
    <add key="oauth" value="YOUR_PASSWORD_HERE"/>
  </appSettings>
</configuration>
```
